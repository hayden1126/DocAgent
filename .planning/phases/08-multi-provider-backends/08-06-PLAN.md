- `litellm.completion(model="anthropic/claude-sonnet-4-6", messages=[...], tools=[...], mock_response="literal text to return")` returns a synthetic ModelResponse whose `choices[0].message.content == "literal text to return"` and whose `usage.prompt_tokens=10, usage.completion_tokens=20` (LiteLLM's mock defaults).
- mock_response does NOT exercise the tool-use loop (no tool_calls in the synthetic response) — that's fine for snapshot stability because we want a SINGLE deterministic output, not a multi-turn dance.
- The snapshot is testing the orchestration shell: backend selected → run() drives one turn → response.content surfaces → cost_usd attached (Tier 3 may fire since LiteLLM's price table may not have the mock; that's expected and pinned in the test).

```python
# tests/golden/test_litellm_backend_snapshot.py
import os
import pytest

litellm_installed = pytest.importorskip("litellm", reason="litellm not installed; skip multi-provider snapshot")

from pathlib import Path
from docagent.backends.base import GenerationRequest
from docagent.backends.litellm_backend import LiteLLMBackend

FIXTURE = Path("tests/golden/fixtures/tinylib_ts")
SNAPSHOT = Path("tests/golden/snapshots/tinylib_ts_litellm_readme.md")

# Recorded LLM output — committed to git. Multiline triple-quoted string OR
# read from SNAPSHOT path. Mock response simulates the model's README output
# with proper <!-- ground: --> citations.
MOCK_README = """# tinylib_ts

A tiny TypeScript library used as a DocAgent test fixture. <!-- ground: package.json:1-5 -->

## Usage

Import the public API from the main entry point. <!-- ground: src/index.ts:1-10 -->
"""

def test_litellm_backend_snapshot(tmp_path):
    backend = LiteLLMBackend(model="anthropic/claude-sonnet-4-6")
    request = GenerationRequest(
        artifact_id="readme",
        prompt="Generate a README.md for this tinylib_ts library.",
        repo_root=FIXTURE.resolve(),
    )
    # Monkeypatch litellm.completion to use mock_response.
    # mock_response is a LiteLLM SDK kwarg, not a global — we route through
    # a small wrapper that injects it.
    import litellm
    original = litellm.completion
    def _mock_completion(**kwargs):
        kwargs["mock_response"] = MOCK_README
        return original(**kwargs)
    litellm.completion = _mock_completion
    try:
        response = backend.run(request)
    finally:
        litellm.completion = original

    actual = response.content
    if os.environ.get("UPDATE_SNAPSHOTS"):
        SNAPSHOT.parent.mkdir(parents=True, exist_ok=True)
        SNAPSHOT.write_text(actual, encoding="utf-8")
        pytest.skip(f"Snapshot updated at {SNAPSHOT}")
    expected = SNAPSHOT.read_text(encoding="utf-8")
    assert actual == expected
```

Note: monkeypatching `litellm.completion` globally inside the test is OK
for a single test; if multiple snapshot tests land later, refactor into
a fixture. The `mock_response` parameter must be passed via the kwargs
that LiteLLMBackend passes to completion() — the wrapper handles this.
